https://atcoder.jp/contests/abc448/tasks/abc448_d

我刚开始的做法是dfs，只要遇到满足条件的点，就把他bfs,把它的子树都设为ok，时间复杂度 ： O(n logn)

* 这里提供一种更优做法

```c++
void solve() {
    int n;
    cin>>n;

    vector<int> a(n+1);
    for (int i=1;i<=n;i++) cin>>a[i];

    vector<vector<int>> adj(n+1);
    for (int i=1;i<n;i++) {
        int u,v;
        cin>>u>>v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }


    vector<int> ans(n+1);
    map<int,int> cnt;//记录当前路径不同值的数量
    //d记录深度
    auto dfs = [&](auto &&self,int u,int p,int d)->void {
        cnt[a[u]]++;
        ans[u]=(d==cnt.size());
        for (auto v:adj[u]) {
            if (v==p) continue;
            self(self,v,u,d+1);
        }

        cnt[a[u]]--;
        if (!cnt[a[u]]) cnt.erase(a[u]);
    };

    dfs(dfs,1,0,1);
    for (int i=1;i<=n;i++) {
        if (!ans[i]) cout<<"Yes\n";
        else cout<<"No\n";
    }

}
```
