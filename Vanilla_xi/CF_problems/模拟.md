# 网格

https://codeforces.com/contest/2207/problem/C

O(n^2)

* 我们先计算出单独的一个排水口的排水量 ，只能接受比他高的，我们分段计算 pre,suf ，a[i]=h-a[i];
  * 都是从i开始，往前/后走，tmp=a[i], 维护tmp=min(tmp,a[i]), 并用tot+=tmp,
* 对于两个排水口，我们需要减去重复的， 对于区间 [i,j], 两个排水口位于i,j, 其实我们只需减去 cnt[x],x 为区间最高处



```c++
void solve() {
    i64 n,h;
    cin>>n>>h;
    vector<i64> a(n+1);
    for (int i=1;i<=n;i++) {
        cin>>a[i];
        a[i]=h-a[i];//首先，上面的水必定可以
    }

    i64 mx=0;
    vector<i64> pre(n+1),suf(n+1);
    for (int i=1;i<=n;i++) {
        i64 tot=0,tmp=infl;
        for (int j=i;j>=1;j--) {
            tmp=min(tmp,a[j]);//只能再接受比他高的
            tot+=tmp;
        }
        pre[i]=tot;
    }

    for (int i=n;i>=1;i--) {
        i64 tot=0,tmp=infl;
        for (int j=i;j<=n;j++) {
            tmp=min(tmp,a[j]);
            tot+=tmp;
        }
        suf[i]=tot;
    }

    vector<i64> cnt(n+1);
    for (int i=1;i<=n;i++) {
        cnt[i]=pre[i]+suf[i]-a[i];
    }

    for (int i=1;i<=n;i++) {
        int id=i;
        for (int j=i;j<=n;j++) {
            if (a[j]<a[id]) id=j;
            mx=max(mx,cnt[i]+cnt[j]-cnt[id]);
        }
    }

    cout<<mx<<'\n';

}
```
